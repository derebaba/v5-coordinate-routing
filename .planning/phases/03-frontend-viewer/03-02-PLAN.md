      const shareUrl = data.url || "";

        status.innerHTML = "";

        const link = document.createElement("a");

        copyBtn.style.marginLeft = "8px";

          try {
            await navigator.clipboard.writeText(shareUrl);
            copyBtn.textContent = "Copied!";
            copyBtn.style.color = "#22573b";
            setTimeout(() => {
              copyBtn.textContent = "Copy Link";
              copyBtn.style.color = "";
            }, 2000);
          } catch {

            copyBtn.style.color = "var(--danger, #c53e3e)";
          }
        });

        status.appendChild(link);

      }
    }

**Key implementation notes:**
- `data.url` is the share URL returned by POST /documents (now points to `/viewer/{id}` per Plan 01)
- Link uses `createElement("a")` with `textContent` — NO innerHTML with API data (XSS prevention per UI-SPEC security contract)
- Copy button uses existing `button.ghost` CSS class from styles.css (line 389)
- Success feedback: "Copied!" with color `#22573b` for 2 seconds, then revert to "Copy Link"
- Failure feedback: "Copy failed" with `var(--danger)` color — stays until next click
- Fallback: if `data.url` is empty/missing, shows plain "Uploaded successfully." text
- `status.innerHTML = ""` is safe here — it clears the element before appending new children (not injecting API data)

    (function () {
      "use strict";

      let documentData = null;

      const loadingEl = document.getElementById("loading");
      const errorEl = document.getElementById("error");
      const contentEl = document.getElementById("content");
      const actionsEl = document.getElementById("actions");

        contentEl.hidden = true;

      function buildSchoolsTable(schools) {
        var wrap = document.createElement("div");
        wrap.className = "table-wrap";

        var table = document.createElement("table");
        var thead = document.createElement("thead");
        var headerRow = document.createElement("tr");

        ["Name", "District", "Type"].forEach(function (text) {
          var th = document.createElement("th");
          th.textContent = text;
          headerRow.appendChild(th);
        });

        thead.appendChild(headerRow);
        table.appendChild(thead);

        var tbody = document.createElement("tbody");
        schools.forEach(function (school) {
          var tr = document.createElement("tr");

          tdDistrict.textContent = school.district || "";
          tr.appendChild(tdDistrict);

          var tdType = document.createElement("td");
          tdType.textContent = school.schoolType || "";
          tr.appendChild(tdType);

          tbody.appendChild(tr);
        });

        table.appendChild(tbody);
        wrap.appendChild(table);
        return wrap;
      }

      function renderDocument(data) {
        loadingEl.hidden = true;
        contentEl.hidden = false;
        actionsEl.hidden = false;

        if (data.cities && data.cities.length > 0 && data.cities[0].name) {
          document.title = "Shared: " + data.cities[0].name + " \u2014 Fieldwork Scheduler";
        }

        data.cities.forEach(function (city) {
          var section = document.createElement("div");
          section.className = "panel";

          var h2 = document.createElement("h2");
          h2.textContent = city.name || "Unnamed City";
          section.appendChild(h2);

          var stats = document.createElement("p");
          stats.className = "muted";
          var schoolCount = (city.schools && city.schools.length) || 0;
          var researcherCount = (city.researchers && city.researchers.length) || 0;
          var dayCount = (city.dayPlans && city.dayPlans.length) || 0;
          stats.textContent = schoolCount + " schools \u00B7 " + researcherCount + " researchers \u00B7 " + dayCount + " working days";
          section.appendChild(stats);

          if (city.schools && city.schools.length > 0) {
            section.appendChild(buildSchoolsTable(city.schools));
          } else {
            var empty = document.createElement("p");
            empty.className = "muted";
            empty.textContent = "No schools in this workspace.";
            section.appendChild(empty);
          }

          contentEl.appendChild(section);
        });
      }

      function downloadJson() {
        if (!documentData) return;
        var blob = new Blob([JSON.stringify(documentData, null, 2)], { type: "application/json" });
        var url = URL.createObjectURL(blob);
        var a = document.createElement("a");
        a.href = url;
        a.download = "shared-document-" + docId + ".json";
        a.click();
        URL.revokeObjectURL(url);
      }

      async function loadDocument() {
        try {
          var res = await fetch("/documents/" + docId);

          if (!res.ok) {
            if (res.status === 404) {
              showError("Document not found. It may have been removed or the link may be incorrect.");
              return;

            showError("Error loading document (status " + res.status + "). Please try again later.");
            return;

            showError("This document uses an unsupported format and cannot be displayed.");
            return;

            return;

        }

**CRITICAL implementation rules (from UI-SPEC Security Contracts):**
- ALL data rendering uses `textContent` or `createElement` — NEVER `innerHTML` with API data
- The `docId` from URL is only used in the fetch URL and download filename — never inserted into HTML
- Fetch uses relative path `/documents/${docId}` — same-origin, no CORS issues
- Error element has `role="alert"` for screen reader accessibility
- All buttons are `<button>` elements (not div/span) for keyboard accessibility
- Focus styles use `outline: 2px solid #9cc9a8; outline-offset: 1px` matching existing app pattern

**Error handling (from UI-SPEC Copywriting Contract):**
- 404: "Document not found. It may have been removed or the link may be incorrect."
- Server error: "Error loading document (status {code}). Please try again later."
- Network failure: "Failed to load document. Please check your connection and try again."
- Malformed data: "This document uses an unsupported format and cannot be displayed."

**Rendering per city:**
- City name as `<h2>` heading
